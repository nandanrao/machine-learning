using Distributions, Gadfly, DataFrames

scale_and_center(p) = (p - mean(p))/var(p)

function project(n = 5)
    p = rand(Normal(), 2, n) * eye(n)
end

function compare_plots(n = 100)
    s = scale_and_center(project(n))
    r = rand(Normal(), 2, n)
    c = DataFrame([s' repeat(["projected"], outer = n); r' repeat(["normal"], outer = n)])
    plot(c, x = "x1", y = "x2", xgroup="x3", Geom.subplot_grid(Geom.point))
end

compare_plots(10)

compare_plots(100)

compare_plots(10000)
